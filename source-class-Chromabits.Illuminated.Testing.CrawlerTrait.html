 329:      *
 330:      * @return bool
 331:      */
 332:     protected function hasLink($text, $url = null)
 333:     {
 334:         $links = $this->crawler->selectLink($text);
 335: 
 336:         if ($links->count() == 0) {
 337:             return false;
 338:         }

 340:         // If the URL is null, we assume the developer only wants to find a link

 345:         }

 347:         $url = $this->addRootToRelativeUrl($url);

 349:         foreach ($links as $link) {

 351:                 return true;
 352:             }
 353:         }
 354: 
 355:         return false;
 356:     }
