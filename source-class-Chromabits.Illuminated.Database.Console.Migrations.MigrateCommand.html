 49:     /**
 50:      * Construct an instance of a MigrateCommand.
 51:      *
 52:      * @param Migrator $migrator
 53:      */
 54:     public function __construct(Migrator $migrator)
 55:     {
 56:         parent::__construct();
 57:

 59:     }
 60: 
 61:     /**
 62:      * Execute the console command.
 63:      *
 64:      */
 65:     public function fire()
 66:     {
 67:         if (!$this->confirmToProceed()) {
 68:             return;
 69:         }
 70: 
 71:         $this->prepareDatabase();
 72: 
 73:         // The pretend option can be used for "simulating" the migration and
 74:         // grabbing the SQL queries that would fire if the migration were to be
 75:         // run against a database for real, which is helpful for double checking
 76:         // migrations.
 77:         $pretend = $this->input->getOption('pretend');
 78: 
 79:         $path = $this->getMigrationPath();
 80: 
 81:         $this->migrator->run($path, $pretend);
 82: 
 83:         // Once the migrator has run we will grab the note output and send it
 84:         // out to the console screen, since the migrator itself functions
 85:         // without having any instances of the OutputInterface contract passed
 86:         // into the class.
 87:         foreach ($this->migrator->getNotes() as $note) {
 88:             $this->output->writeln($note);
 89:         }
 90: 
 91:         // Finally, if the "seed" option has been given, we will re-run the
 92:         // database seed task to re-populate the database, which is convenient
 93:         // when adding a migration and a seed at the same time, as it is only
 94:         // this command.
 95:         if ($this->input->getOption('seed')) {
 96:             $this->call('db:seed', ['--force' => true]);
 97:         }
 98:     }
 99: 
100:     /**
101:      * Prepare the migration database for running.
102:      *
103:      */
104:     protected function prepareDatabase()
105:     {
106:         $this->migrator->setConnection($this->input->getOption('database'));
107: 
108:         if (!$this->migrator->repositoryExists()) {
109:             $options = ['--database' => $this->input->getOption('database')];
110: 
111:             $this->call('migrate:install', $options);
112:         }
113:     }
